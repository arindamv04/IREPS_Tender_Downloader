function modifyDays(dateStr, days) {
var parts = dateStr.split('/');
var day = parseInt(parts[0], 10);
var month = parseInt(parts[1], 10);
var year = parseInt(parts[2], 10);

var date = new Date(year, month - 1, day); // Months are 0-indexed in JavaScript Date
date.setDate(date.getDate() + days);

var newDay = String(date.getDate()).padStart(2, '0');
var newMonth = String(date.getMonth() + 1).padStart(2, '0');
var newYear = date.getFullYear();

return newDay + '/' + newMonth + '/' + newYear;
}

function modifyMonths(dateStr, months) {
var parts = dateStr.split('/');
var day = parseInt(parts[0], 10);
var month = parseInt(parts[1], 10);
var year = parseInt(parts[2], 10);

var date = new Date(year, month - 1, day);
var originalDay = date.getDate();
date.setMonth(date.getMonth() + months);

// Handling month overflow
if (date.getDate() !== originalDay) {
date.setDate(0); // Last day of the previous month
}

var newDay = String(date.getDate()).padStart(2, '0');
var newMonth = String(date.getMonth() + 1).padStart(2, '0');
var newYear = date.getFullYear();

return newDay + '/' + newMonth + '/' + newYear;
}
